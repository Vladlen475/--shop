<script>

let products = [
    {name:"MSI Katana 17 HX", price:159990, type:"gaming", img:"laptop.png"},
    {name:"Quantum Air", price:89990, type:"office", img:"https://via.placeholder.com/300x200"}
];

let cart = JSON.parse(localStorage.getItem("cart")) || [];
let fav = [];

function render(list){
    let html="";
    list.forEach((p,i)=>{
        html+=`
        <div class="card">
            <img src="${p.img}">
            <h3>${p.name}</h3>
            <p>${p.price} ₽</p>
            <button class="btn" onclick="addToCart(${i})">Купить</button>
            <div class="fav" onclick="addFav()">❤️</div>
        </div>`;
    });
    document.getElementById('products').innerHTML=html;
}

function addToCart(i){
    cart.push(products[i]);
    localStorage.setItem("cart", JSON.stringify(cart));
    updateCart();
    toast("Добавлено в корзину");
}

function updateCart(){
    document.getElementById('cart-count').innerText=cart.length;

    let total=0, html="";
    cart.forEach((p,i)=>{
        total+=p.price;
        html+=`<p>${p.name} - ${p.price} ₽ 
        <button onclick="remove(${i})">❌</button></p>`;
    });

    document.getElementById('cart-items').innerHTML=html;
    document.getElementById('total').innerText="Итого: "+total+" ₽";
}

function remove(i){
    cart.splice(i,1);
    localStorage.setItem("cart", JSON.stringify(cart));
    updateCart();
}

function toggleCart(){
    document.getElementById('cart').classList.toggle('open');
}

function addFav(){
    fav++;
    document.getElementById('fav-count').innerText=fav;
}

function search(q){
    let res = products.filter(p=>p.name.toLowerCase().includes(q.toLowerCase()));
    render(res);
}

function toast(text){
    let t=document.getElementById('toast');
    t.innerText=text;
    t.classList.add('show');
    setTimeout(()=>t.classList.remove('show'),2000);
}

render(products);
updateCart();

</script>
